(define (mushroomUpdate lom bullet)
  (cond
       [(empty? lom) empty]
       [(cons? lom)(if (and (same-posn? (mushroom-posn (first lom)) (bullet-posn bullet)) (bullet-present? bullet))
                        (if
                           (or (symbol=? (mushroom-color (first lom)) MUSHROOM-5-C)
                               (symbol=? (mushroom-color (first lom)) MUSHROOM-4-C)
                              (symbol=? (mushroom-color (first lom)) MUSHROOM-3-C)
                              (symbol=? (mushroom-color (first lom)) MUSHROOM-2-C))
                               (cons (cond
                          [(symbol=? (mushroom-color (first lom)) MUSHROOM-5-C)
                           (make-mushroom (mushroom-posn (first lom)) MUSHROOM-4-C)]
                          [(symbol=? (mushroom-color (first lom)) MUSHROOM-4-C)
                           (make-mushroom (mushroom-posn (first lom)) MUSHROOM-3-C)]
                          [(symbol=? (mushroom-color (first lom)) MUSHROOM-3-C)
                           (make-mushroom (mushroom-posn (first lom)) MUSHROOM-2-C)]
                          [(symbol=? (mushroom-color (first lom)) MUSHROOM-2-C)
                           (make-mushroom (mushroom-posn (first lom)) MUSHROOM-1-C)])
                    (mushroomUpdate (rest lom) bullet))
                               (mushroomUpdate (rest lom) bullet))
                        (cons (first lom) (mushroomUpdate (rest lom) bullet)))]))
